> 万能公式 dp
>
> ```
> int len = prices.length;
>         if (len <= 1) {
>             return 0;
>         }
>         int[][] dp = new int[len][2];
>         for (int i = 0; i < len; i++) {
>             if (i == 0) {
>                 dp[0][0] = 0;
>                 dp[0][1] = -prices[0];
>                 continue;
>             }
>
>             dp[i][0] = Math.max(dp[i - 1][0], dp[i - 1][1] + prices[i] - fee);
>             dp[i][1] = Math.max(dp[i - 1][1], dp[i - 1][0] - prices[i]);
>         }
>         return dp[len - 1][0];
> ```
